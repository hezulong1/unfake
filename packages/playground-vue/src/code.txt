<script setup lang="ts">
import { onMounted, ref } from 'vue';
import { Repl, useStore, File } from '@vue/repl';
import Editor from '@vue/repl/monaco-editor';
import compCode from './components/FlexibleScreen.vue?raw';

const compFile = new File(
  'src/FlexibleScreen.vue',
  compCode,
);
const store = useStore({
  files: ref({
    [compFile.filename]: compFile,
  }),
});

// 3. 在组件挂载时，将你的代码写入 Store 的主文件（App.vue）中
onMounted(() => {
  // @vue/repl 默认的主入口文件叫 'App.vue'
  // store.state.files 是一个键值对对象
  if (store.files['src/App.vue']) {
    store.files['src/App.vue'].code = `
      <script setup lang="ts">
        import FlexibleScreen from './FlexibleScreen.vue'
      <\/script>
      <template>
        <FlexibleScreen :natural-width="1920" :natural-height="1080" preserve-aspect-ratio>
          <div style="background: red; width: 100%; height: 100%" />
        </FlexibleScreen>
      </template>
    `;
  }
});

// store.addFile(compFile);

Object.defineProperty(window, 'store', {
  value: store,
});
</script>

<template>
  <Repl
    :editor="Editor"
    :store
    :preview-options="{
      headHTML: `<link rel='stylesheet' href='/reset.css' />`,
    }"
  />
</template>
